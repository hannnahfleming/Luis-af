<script>

    // References

    const menu = document.getElementById('menu');

    const page1 = document.getElementById('page1');

    const page2 = document.getElementById('page2');

    const fade = document.getElementById('fade');


    const single = document.getElementById('single');

    const multi = document.getElementById('multi');

    const quit = document.getElementById('quit');


    // Floating text lists

    const reasons = [

        "You are kind",

        "You are thoughtful",

        "You treat me so well",

        "You are handsome",

        "You are hardworking",

        "You are caring",

        "There are a million other reasons but I'll keep it short on here"

    ];


    const excited = [

        "Travel with you",

        "Go hiking with you",

        "Try new restaurants with you",

        "Explore new places with you",

        "Literally anything with you"

    ];


    // Function to create floating texts

    function createFloatingText(container, texts) {

        texts.forEach(text => {

            const el = document.createElement('div');

            el.className = 'floating-text';

            el.textContent = text;


            // Random initial position

            el.style.left = Math.random() * (window.innerWidth - 200) + 'px';

            el.style.top = Math.random() * (window.innerHeight - 50) + 'px';


            // Random velocity

            el.vx = (Math.random() - 0.5) * 2;

            el.vy = (Math.random() - 0.5) * 2;


            container.appendChild(el);

        });


        function animate() {

            const children = container.getElementsByClassName('floating-text');

            for (let c of children) {

                let left = parseFloat(c.style.left);

                let top = parseFloat(c.style.top);


                // Bounce

                if(left < 0 || left > window.innerWidth - 200) c.vx *= -1;

                if(top < 0 || top > window.innerHeight - 50) c.vy *= -1;


                c.style.left = left + c.vx + 'px';

                c.style.top = top + c.vy + 'px';

            }

            requestAnimationFrame(animate);

        }

        animate();

    }


    // Button actions

    single.addEventListener('click', () => {

        menu.style.display = 'none';

        page1.style.display = 'block';

        createFloatingText(page1, reasons);

    });


    multi.addEventListener('click', () => {

        menu.style.display = 'none';

        page2.style.display = 'block';

        createFloatingText(page2, excited);

    });


    quit.addEventListener('click', () => {

        fade.style.display = 'block';

        setTimeout(() => { fade.style.opacity = 1; }, 50);

    });
